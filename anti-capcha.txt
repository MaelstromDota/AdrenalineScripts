var Captcha:string;
begin
  while true do
    begin
      delay(500) ;
        if Captcha <> Engine.DlgText then
          begin
            Captcha := Engine.DlgText; // Проверка на диалог
              if (Pos('защиты от ботов', Captcha) <> 0) then // Проверка на содержимое диалогового окна
                PlaySound('Sounds/PlayerAlarm.wav',false); // Звук который будет проигрываться
                Delay(1800);
          end;
    end;
end.
