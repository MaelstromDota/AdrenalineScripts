uses SysUtils;
var tmpDialog,tmpString:string;
var posMath, num1, num2, resultNum, count:integer;
begin
  while Engine.Status = lsOnline do begin
    tmpDialog := Engine.DlgText;
    if (Pos('будет', tmpDialog) <> 0) then begin
      tmpString:='';
      tmpDialog:='';
      tmpDialog:=Engine.DlgText;
      posMath:=pos('будет',tmpDialog);
      posMath:=posMath+6;
      while not (tmpDialog[posMath] = ' ') do begin
        tmpString:=tmpString + tmpDialog[posMath];
        posMath:=posMath + 1;
      end;
      num1:=StrToInt(tmpString);
      tmpString:='';
      posMath:=posMath + 3;
      while not (tmpDialog[posMath] = '?') do begin
        tmpString:=tmpString + tmpDialog[posMath];
        posMath:=posMath + 1;
      end;
      num2:=StrToInt(tmpString);
      resultNum:=num1 + num2;
      print(IntToStr(resultNum));
    end;
    Delay(888);
  end;
end.
